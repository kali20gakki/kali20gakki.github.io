FPGA UART模块设计

采用TOP_DOWN模式，TOP为uart_module.v。子模块有三个，uart_divclk.v, uart_rx.v, uart_tx.v 。

uart_module.v 文件如下，负责各模块的实例化以及连线。

/* 串口TOP模块 */
module uart_module(sys_clk,sys_rst_n,uart_rx,uart_tx);

input sys_clk;
input sys_rst_n;

input uart_rx;    
output uart_tx;

//wire define
wire clk_BPS;
wire [7:0] data_inter;
wire sig_inter;


/* 子模块实例化 */
uart_divclk u_uart_divclk(
   .clk50    (sys_clk),
     .rst_n    (sys_rst_n),
     .clkout   (clk_BPS)
);


uart_rx u_uart_rx(
   .clk      (clk_BPS),
     .rst_n    (sys_rst_n),
     .rx       (uart_rx),
     .dataout  (data_inter),
     .rdsig    (sig_inter),
     .dataerror(), //留空表示不用
     .frameerror()
);


uart_tx u_uart_tx(
   .clk      (clk_BPS),
     .rst_n    (sys_rst_n),
     .datain   (data_inter),
     .wrsig    (sig_inter),
     .idle     (),
     .tx       (uart_tx),
);

endmodule
